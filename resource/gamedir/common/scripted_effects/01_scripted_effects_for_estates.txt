#Scripted effects for estates


#Clergy
add_estate_church_loyalty_effect = {
	if = {
		limit = {
			has_estate = estate_church
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}
	}
	else_if = {
		limit = {
			religion = hinduism
			has_estate = estate_brahmins
		}
		add_estate_loyalty = {
			estate = estate_brahmins
			loyalty = 10
		}
	}
}
reduce_estate_church_loyalty_effect = {
	if = {
		limit = {
			has_estate = estate_church
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
	}
	else_if = {
		limit = {
			religion = hinduism
			has_estate = estate_brahmins
		}
		add_estate_loyalty = {
			estate = estate_brahmins
			loyalty = -10
		}
	}
}
