estate_dhimmi_disaster = {
	potential = {
		has_estate = estate_dhimmi
		estate_influence = {
			estate = estate_dhimmi
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = dhimmi_estate_triggered }
			had_country_flag = {
				flag = dhimmi_estate_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_dhimmi
			influence = 100
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_dhimmi
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 1
	}
	
	progress = {	
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_dhimmi
				influence = 100
			}
		}		
	}
	
	can_end = {
		always = yes
	}

	on_start = estate_disasters.9
	
	on_start_effect = {
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			custom_tooltip = estate_dhimmi_disaster_tooltip_cossacks
		}
		else = {
			custom_tooltip = estate_dhimmi_disaster_tooltip
		}
	}
	
	on_monthly = {
	}
}
