estate_brahmins_disaster = {
	potential = {
		has_dlc = "Dharma"
		has_estate = estate_brahmins
		estate_influence = {
			estate = estate_brahmins
			influence = 80
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_brahmins
			influence = 100
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_brahmins
					influence = 100
				}
			}
		}
	}
	
	down_progress = {
		factor = 1
	}
	
	progress = {
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_brahmins
				influence = 100
			}
		}		
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_BRAHMINS
			OR = {
				NOT = { has_country_flag = brahmin_estate_in_power }
				has_country_flag = noble_estate_in_power
				has_country_flag = church_estate_in_power
			}
		}
	}
	
	modifier = {	
		technology_cost = 0.2
		idea_cost = 0.2
		tolerance_heretic = -2
		tolerance_heathen = -2
	}

	on_start = estate_disasters.12
	on_end = estate_disasters.13
	
	on_monthly = {
	}
}
